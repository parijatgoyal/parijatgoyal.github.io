#include #include // Function to dynamically allocate a 2D int array int** allocate_array(int M, int N) { int** array = (int**)malloc(M * sizeof(int*)); for (int i = 0; i < M; i++) { array[i] = (int*)malloc(N * sizeof(int)); } return array; } // Function to read an image from user input void read_image(int*** image, int* row_size, int* column_size) { // Read the row size and column size of the image printf("Enter the row size of the image: "); scanf("%d", row_size); printf("Enter the column size of the image: "); scanf("%d", column_size); // Allocate memory for the image *image = allocate_array(*row_size, *column_size); // Read the image data printf("Enter the image data (0 or 1):\n"); for (int i = 0; i < *row_size; i++) { for (int j = 0; j < *column_size; j++) { scanf("%d", &((*image)[i][j])); } } } // Function to print the content of an image void print_image(int** image, int row_size, int column_size) { printf("Image content:\n"); for (int i = 0; i < row_size; i++) { for (int j = 0; j < column_size; j++) { printf("%d ", image[i][j]); } printf("\n"); } } // Function to get the boundary image int** get_boundary_image(int** image, int row_size, int column_size) { int** boundary_image = allocate_array(row_size, column_size); for (int i = 0; i < row_size; i++) { for (int j = 0; j < column_size; j++) { if (image[i][j] == 1) { // Check if any neighbor is a background point if ((i > 0 && image[i - 1][j] == 0) || (i < row_size - 1 && image[i + 1][j] == 0) || (j > 0 && image[i][j - 1] == 0) || (j < column_size - 1 && image[i][j + 1] == 0)) { boundary_image[i][j] = 1; // Object point is a boundary point } else { boundary_image[i][j] = 0; // Object point is not a boundary point } } else { boundary_image[i][j] = 0; // Background point } } } return boundary_image; } int main() { int** image; // Print the input image printf("Input Image:\n"); print_image(image, row_size, column_size); // Print the boundary image printf("Boundary Image:\n"); print_image(boundary_image, row_size, column_size);
